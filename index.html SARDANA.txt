    <script>
        function sendMessage() {
            const input = document.getElementById('user-input');
            const chatWindow = document.getElementById('chat-window');
            
            if (input.value.trim() === "") return;

            // Tambahkan pesan user ke layar
            const userMsg = `<div class="flex justify-end">
                                <div class="bg-gray-200 text-black p-3 rounded-lg max-w-[80%]">
                                    ${input.value}
                                </div>
                             </div>`;
            chatWindow.innerHTML += userMsg;
            
            // Scroll ke bawah
            input.value = "";
            chatWindow.scrollTop = chatWindow.scrollHeight;

            // Simulasi respon (Nanti dihubungkan ke Backend/API)
            setTimeout(() => {
                const aiResponse = `<div class="flex items-start">
                                        <div class="bg-stah-red text-white p-3 rounded-lg max-w-[80%]">
                                            <p class="font-bold text-xs mb-1">SADANA</p>
                                            Terima kasih atas pertanyaannya. Saya sedang memproses informasi mengenai STAH Dharma Sentana untuk Anda.
                                        </div>
                                    </div>`;
                chatWindow.innerHTML += aiResponse;
                chatWindow.scrollTop = chatWindow.scrollHeight;
            }, 1000);
        }
    </script>
